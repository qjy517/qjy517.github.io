#include <iostream>
#include <string>
using namespace std;
//全部运算符的重载
class Person
{
    public:
    string *m_name;
    int m_age;
    //构造函数
    Person(int age,string name)
    {
        m_age=age;
        m_name=new string (name);
    }
    //默认构造函数
    Person():m_age(0),m_name(nullptr){}
    //因为类中有指针，要有拷贝构造函数进行深拷贝
    Person(const Person&p)
    {
        this->m_age=p.m_age;
        this->m_name=new string (*(p.m_name));
    }
    //析构函数
    ~Person()
    {
        if (this->m_name!=nullptr)
        {
            delete this->m_name;
            this->m_name=nullptr;
        }
    }
    //赋值重载
    Person&operator=(const Person &p)
    {
        this->m_age=p.m_age;
        this->m_name=new string (*(p.m_name));//深拷贝
        return *this;
    }
};
//+的重载
//返回类型不是引用，因为要返回的时临时变量
Person operator+(const Person &p1,const Person &p2)
{
    Person p;
    p.m_age=p1.m_age+p2.m_age;
    p.m_name=new string (*(p1.m_name)+*(p2.m_name));
    return p;
}
//>>的 重载
//返回的是引用，便于链式使用，流只能是引用传递
istream &operator>>(istream &is,Person &p)
{
    string str;
    cin>>p.m_age>>str;
    p.m_name=new string (str);
    return is;
}
//<<的重载
ostream &operator<<(ostream &os,Person p)
{
    os<<p.m_age<<" "<<*(p.m_name);
    return os;
}
//前置++
Person &operator++(Person &p)
{
    p.m_age++;
    return p;
}
//赋值重载在类内
//后置++
Person operator++(Person &p,int)
{
    Person temp;
    //要提前写出=的重载
    temp=p;
    p.m_age++;
    return temp;
}
//==的重载
bool operator==(const Person &p1,const Person &p2)
{
    return (p1.m_age==p2.m_age&&*(p1.m_name)==*(p2.m_name));
}
//！=的重载
bool operator!=(const Person &p1,const Person &p2)
{
    return (p1.m_age!=p2.m_age||*(p1.m_name)!=*(p2.m_name));
}
void test01()
{
    Person p;
    Person temp(14,"mike");
    p=temp;
    cout<<p<<endl;
}
void test02()
{
    Person p1(13,"Amy");
    Person p2(12,"Mike");
    Person p3=p1+p2;
    cout<<p3<<endl;
    cout<<p1+p2<<endl;
}
void test03()
{
    Person p;
    cin>>p;
    cout<<p<<endl;
}
int main()
{
    test01();
    test02();
    test03();
    return 0;
}
