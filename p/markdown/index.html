 #include <stdlib.h>
 void hanoi(int n,char F,char E,char T);
 void move(char F,char E);
 int main()
 {
    int n;
    printf("Input the number of disks:");
    scanf("%d",&n);
    printf("Steps of moving %d disks from A to C by means of B:\n",n);
    hanoi(n,'A','C','B');
     return 0;
 }
 void hanoi(int n,char F,char E,char T)
 {
     if (n==1)
     {
         move(F,E);
     }
 //后面三个语句要用{}括起来，因为不括起来，在基准问题即n=1，执行完之后
 //不会返回到上一个主调函数，会继续运行hanoi
 //还要注意，这里递归调用hanoi的参数应为主调函数的F,E,T,不是'A''B''C'
     else
     {
     hanoi(n-1,F,T,E);
     move(F,E);
     hanoi(n-1,T,E,F);
     }
 }
 void move(char F,char E)
 {
     printf("%c->%c\n",F,E);
 }
